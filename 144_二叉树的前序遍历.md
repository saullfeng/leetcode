 ### 递归

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> res =new ArrayList<>();
        indor(root,res);
        return res;

    }
    private void indor(TreeNode root,List<Integer> res){
        if(root ==null)return;
        res.add(root.val);
        indor(root.left,res);
        indor(root.right,res);
    }
}
```

### 迭代

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<Integer> preorderTraversal(TreeNode root) {
        List<Integer> res =new ArrayList<>();
        if(root==null) return null;
        Deque<TreeNode> stk =new LinkedList<>();
        while(root!=null||!stk.isEmpty()){
            while(root!=null){
            res.add(root.val);
            stk.push(root);
            root=root.left;
            }
            root=stk.pop();
            root=root.right;
         
        }
        return res;
    
    }
}
```
